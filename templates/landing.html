import asyncio, aiohttp
from bs4 import BeautifulSoup

async def fetch(url, retries=3):
  async with aiohttp.ClientSession() as s:
    for i in range(retries):
      try:
        r = await s.get(url)
        return await r.text()
      except Exception:
        await asyncio.sleep(2**i)
