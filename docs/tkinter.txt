Tkinter Documentation

Tkinter is Python's standard GUI (Graphical User Interface) package.

Basic Window:
import tkinter as tk
from tkinter import ttk

# Create main window
root = tk.Tk()
root.title('My Application')
root.geometry('400x300')
root.mainloop()

Widgets:
# Label
label = tk.Label(root, text='Hello, World!', font=('Arial', 12))
label.pack(pady=10)

# Button
def on_click():
    print('Button clicked!')

button = tk.Button(root, text='Click Me', command=on_click)
button.pack(pady=5)

# Entry
entry = tk.Entry(root, width=30)
entry.pack(pady=5)
text = entry.get()

# Text widget
text_widget = tk.Text(root, height=10, width=40)
text_widget.pack(pady=5)
content = text_widget.get('1.0', tk.END)

# Listbox
listbox = tk.Listbox(root)
listbox.pack(pady=5)
listbox.insert(tk.END, 'Item 1')
listbox.insert(tk.END, 'Item 2')
selection = listbox.get(listbox.curselection())

Layout Management:
# pack() - simple layout
widget.pack(side=tk.TOP, fill=tk.X, padx=5, pady=5)

# grid() - table layout
label1 = tk.Label(root, text='Name:')
label1.grid(row=0, column=0, padx=5, pady=5)
entry1 = tk.Entry(root)
entry1.grid(row=0, column=1, padx=5, pady=5)

# place() - absolute positioning
widget.place(x=50, y=50, width=100, height=30)

Canvas Widget:
canvas = tk.Canvas(root, width=400, height=300, bg='white')
canvas.pack()

# Draw shapes
canvas.create_line(10, 10, 100, 100, fill='blue', width=2)
canvas.create_rectangle(50, 50, 150, 100, fill='red', outline='black')
canvas.create_oval(100, 100, 200, 200, fill='green')
canvas.create_text(200, 50, text='Hello Canvas', font=('Arial', 12))

# Handle events
def on_canvas_click(event):
    x, y = event.x, event.y
    canvas.create_oval(x-10, y-10, x+10, y+10, fill='blue')

canvas.bind('<Button-1>', on_canvas_click)

Treeview (Table):
tree = ttk.Treeview(root, columns=('Name', 'Age', 'City'), show='headings')
tree.heading('Name', text='Name')
tree.heading('Age', text='Age')
tree.heading('City', text='City')
tree.pack(fill=tk.BOTH, expand=True)

# Add data
tree.insert('', tk.END, values=('John', 25, 'New York'))
tree.insert('', tk.END, values=('Jane', 30, 'London'))

# Get selection
def on_tree_select(event):
    selected = tree.selection()
    if selected:
        item = tree.item(selected[0])
        values = item['values']
        print(values)

tree.bind('<<TreeviewSelect>>', on_tree_select)

Event Handling:
# Bind events
def on_key_press(event):
    print(f'Key pressed: {event.char}')

def on_mouse_motion(event):
    print(f'Mouse position: {event.x}, {event.y}')

root.bind('<Key>', on_key_press)
root.bind('<Motion>', on_mouse_motion)

# Custom events
def custom_handler(event):
    print('Custom event triggered')

root.bind('<<CustomEvent>>', custom_handler)
root.event_generate('<<CustomEvent>>')

File Dialog Example:
def open_file():
    filename = filedialog.askopenfilename(
        title='Open Text File',
        filetypes=[('Text Files', '*.txt'), ('All Files', '*.*')]
    )
    if filename:
        with open(filename, 'r') as file:
            text_widget.delete('1.0', tk.END)
            text_widget.insert('1.0', file.read())

def save_file():
    filename = filedialog.asksaveasfilename(
        title='Save Text File',
        defaultextension='.txt',
        filetypes=[('Text Files', '*.txt'), ('All Files', '*.*')]
    )
    if filename:
        with open(filename, 'w') as file:
            file.write(text_widget.get('1.0', tk.END))

# Menu buttons
open_btn = tk.Button(root, text='Open', command=open_file)
save_btn = tk.Button(root, text='Save', command=save_file)
open_btn.pack(pady=5)
save_btn.pack(pady=5)

Best Practices:
- Use ttk widgets for modern look
- Organize code with classes for complex apps
- Use proper layout managers (pack/grid/place)
- Handle window closing event
- Use consistent padding and spacing
- Separate UI logic from business logic
- Test on different platforms (Windows/Mac/Linux)
