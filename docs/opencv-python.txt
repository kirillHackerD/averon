========================================
БИБЛИОТЕКА: opencv-python (cv2)
========================================

ТИП: Сторонняя библиотека
ЯЗЫК: Python
ОФИЦИАЛЬНАЯ ДОКУМЕНТАЦИЯ: https://docs.opencv.org/
РЕПОЗИТОРИЙ: https://github.com/opencv/opencv-python

ОПИСАНИЕ:
Python обертка для OpenCV (Open Source Computer Vision Library).
Мощная библиотека для компьютерного зрения, обработки изображений
и машинного обучения.

ОСНОВНЫЕ МОДУЛИ:
- cv2: Основной модуль с функциями
- cv2.aruco: Детекция маркеров ArUco
- cv2.dnn: Глубокое обучение (Deep Neural Networks)
- cv2.face: Распознавание лиц
- cv2.video: Видео обработка
- cv2.viz: 3D визуализация

ОСНОВНЫЕ ОПЕРАЦИИ С ИЗОБРАЖЕНИЯМИ:
- cv2.imread(path, flags) - Чтение изображения
- cv2.imwrite(path, img) - Сохранение изображения
- cv2.imshow(winname, img) - Отображение изображения
- cv2.cvtColor(img, code) - Преобразование цветового пространства
- cv2.resize(img, dsize) - Изменение размера
- cv2.rotate(img, rotateCode) - Поворот изображения
- cv2.flip(img, flipCode) - Отражение изображения
- cv2.threshold(img, thresh, maxval, type) - Пороговая обработка

ЦВЕТОВЫЕ ПРОСТРАНСТВА:
- cv2.COLOR_BGR2RGB: BGR -> RGB
- cv2.COLOR_BGR2GRAY: BGR -> Grayscale
- cv2.COLOR_BGR2HSV: BGR -> HSV
- cv2.COLOR_BGR2LAB: BGR -> LAB

ОСНОВНЫЕ ФУНКЦИИ ОБРАБОТКИ:
- cv2.GaussianBlur(img, ksize, sigma) - Гауссово размытие
- cv2.medianBlur(img, ksize) - Медианное размытие
- cv2.Canny(img, threshold1, threshold2) - Детекция границ Canny
- cv2.Sobel(img, ddepth, dx, dy) - Оператор Собеля
- cv2.Laplacian(img, ddepth) - Оператор Лапласа

РАБОТА С КОНТУРАМИ:
- cv2.findContours(img, mode, method) - Найти контуры
- cv2.drawContours(img, contours, contourIdx, color) - Нарисовать контуры
- cv2.contourArea(contour) - Площадь контура
- cv2.arcLength(contour, closed) - Длина контура
- cv2.boundingRect(points) - Ограничивающий прямоугольник
- cv2.minEnclosingCircle(points) - Минимальная описанная окружность

ГЕОМЕТРИЧЕСКИЕ ПРИМИТИВЫ:
- cv2.line(img, pt1, pt2, color, thickness) - Линия
- cv2.rectangle(img, pt1, pt2, color, thickness) - Прямоугольник
- cv2.circle(img, center, radius, color, thickness) - Круг
- cv2.putText(img, text, org, fontFace, fontScale, color) - Текст

РАБОТА С ВИДЕО:
- cv2.VideoCapture(device) - Открыть камеру или файл
- cap.read() - Чтение кадра
- cap.release() - Освободить камеру
- cv2.VideoWriter(filename, fourcc, fps, frameSize) - Запись видео

РАБОТА С МАСКАМИ И КЕРНАЛАМИ:
- cv2.bitwise_and(src1, src2, mask) - Побитовое И
- cv2.bitwise_or(src1, src2) - Побитовое ИЛИ
- cv2.bitwise_xor(src1, src2) - Побитовое XOR
- cv2.bitwise_not(src) - Побитовое НЕ
- cv2.getStructuringElement(shape, ksize) - Получить структурный элемент
- cv2.erode(src, kernel) - Эрозия
- cv2.dilate(src, kernel) - Расширение
- cv2.morphologyEx(src, op, kernel) - Морфологическая операция

ОСНОВНЫЕ КОНСТАНТЫ:
- cv2.WINDOW_NORMAL: Изменяемое окно
- cv2.WINDOW_AUTOSIZE: Автоматический размер окна
- cv2.THRESH_BINARY: Бинарная пороговая обработка
- cv2.THRESH_BINARY_INV: Инвертированная бинарная пороговая обработка
- cv2.THRESH_OTSU: Метод Оцу
- cv2.RETR_TREE: Иерархия контуров
- cv2.CHAIN_APPROX_SIMPLE: Упрощенное представление контуров

img = cv2.imread('image.jpg')

    print(f'Размер: {img.shape}')
    print(f'Тип: {img.dtype}')

    gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
    
    # Изменение размера
    resized = cv2.resize(img, (640, 480))
    
    # Размытие
    blurred = cv2.GaussianBlur(img, (5, 5), 0)
    
    # Детекция границ
    edges = cv2.Canny(img, 100, 200)
    
    # Сохранение
    cv2.imwrite('output.jpg', edges)
    
    # Отображение
    cv2.imshow('Original', img)
    cv2.imshow('Edges', edges)
    cv2.waitKey(0)
    cv2.destroyAllWindows()

# Детекция лиц
face_cascade = cv2.CascadeClassifier(
    cv2.data.haarcascades + 'haarcascade_frontalface_default.xml'
)

img = cv2.imread('faces.jpg')
gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
faces = face_cascade.detectMultiScale(gray, 1.1, 4)

for (x, y, w, h) in faces:
    cv2.rectangle(img, (x, y), (x + w, y + h), (255, 0, 0), 2)

cv2.imshow('Faces', img)

# Поиск контуров
img = cv2.imread('shapes.jpg')
gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
_, thresh = cv2.threshold(gray, 127, 255, cv2.THRESH_BINARY)
contours, _ = cv2.findContours(thresh, cv2.RETR_TREE, cv2.CHAIN_APPROX_SIMPLE)

for contour in contours:
    area = cv2.contourArea(contour)
    perimeter = cv2.arcLength(contour, True)
    print(f'Area: {area}, Perimeter: {perimeter}')

cv2.drawContours(img, contours, -1, (0, 255, 0), 2)

# Аффинные преобразования
rows, cols = img.shape[:2]

hist = cv2.calcHist([img], [0], None, [256], [0, 256])
cv2.normalize(hist, hist, 0, 255, cv2.NORM_MINMAX)
