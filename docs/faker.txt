ТИП: Сторонняя библиотека
ЯЗЫК: Python
ОФИЦИАЛЬНАЯ ДОКУМЕНТАЦИЯ: https://faker.readthedocs.io/
РЕПОЗИТОРИЙ: https://github.com/joke2k/faker

ОПИСАНИЕ:
Библиотека для генерации поддельных (фейковых) данных.
Позволяет создавать реалистичные имена, адреса, email, номера телефонов,
даты, текст и другие данные для тестирования, заполнения баз данных и т.д.

ОСНОВНЫЕ КЛАССЫ:
- Faker: Основной класс генератора
- Generator: Генератор данных
- Provider: Провайдер данных (можно создавать свои)

ОСНОВНЫЕ МЕТОДЫ:
- faker.name() - Полное имя
- faker.first_name() - Имя

- faker.password() - Пароль
- faker.phone_number() - Номер телефона
- faker.address() - Адрес

- faker.country() - Страна
- faker.postcode() - Почтовый индекс
- faker.date() - Дата
- faker.date_time() - Дата и время
- faker.year() - Год
- faker.month() - Месяц
- faker.day_of_week() - День недели
- faker.text() - Текст
- faker.sentence() - Предложение
- faker.paragraph() - Параграф
- faker.word() - Слово
- faker.words() - Список слов
- faker.uuid4() - UUID
- faker.url() - URL
- faker.domain_name() - Имя домена
- faker.ipv4() - IPv4 адрес
- faker.ipv6() - IPv6 адрес
- faker.mac_address() - MAC адрес
- faker.user_name() - Имя пользователя
- faker.company() - Название компании
- faker.job() - Профессия
- faker.credit_card_number() - Номер кредитной карты
- faker.credit_card_provider() - Провайдер кредитной карты
- faker.currency() - Валюта
- faker.color_name() - Название цвета

- faker.rgb_color() - RGB цвет

from faker import Faker
import pandas as pd

# СОЗДАНИЕ ГЕНЕРАТОРА
fake = Faker()  # Английская локализация
fake_ru = Faker('ru_RU')  # Русская локализация
fake_uk = Faker('uk_UA')  # Украинская локализация

print("=== РУССКАЯ ЛОКАЛИЗАЦИЯ ===")
print(f"Имя: {fake_ru.name()}")
print(f"Адрес: {fake_ru.address()}")
print(f"Телефон: {fake_ru.phone_number()}")
print(f"Email: {fake_ru.email()}")
print(f"Компания: {fake_ru.company()}")
print(f"Профессия: {fake_ru.job()}")
print(f"Текст: {fake_ru.text()[:100]}...")

print("\n=== АНГЛИЙСКАЯ ЛОКАЛИЗАЦИЯ ===")
print(f"Name: {fake.name()}")
print(f"Address: {fake.address()}")
print(f"Phone: {fake.phone_number()}")

print(f"Company: {fake.company()}")

print(f"UUID: {fake.uuid4()}")

print(f"Текст: {fake.text()[:200]}...")

# ГЕНЕРАЦИЯ МАССИВА ДАННЫХ
print("\n=== ГЕНЕРАЦИЯ 10 ПОЛЬЗОВАТЕЛЕЙ ===")
users = []
for _ in range(10):
    user = {
        'id': fake.uuid4(),
        'first_name': fake_ru.first_name(),
        'last_name': fake_ru.last_name(),
        'email': fake.email(),
        'phone': fake_ru.phone_number(),
        'address': fake_ru.address(),
        'city': fake_ru.city(),
        'postcode': fake_ru.postcode(),
        'birth_date': fake.date_of_birth(minimum_age=18, maximum_age=80),
        'job': fake_ru.job(),
        'company': fake_ru.company(),
        'salary': fake.random_int(min=30000, max=150000),
        'is_active': fake.boolean(chance_of_getting_true=80),
        'registered_at': fake.date_time_this_year()
    }
    users.append(user)

# Создание DataFrame
df = pd.DataFrame(users)
print(df.head())

# СОХРАНЕНИЕ В CSV
df.to_csv('fake_users.csv', index=False, encoding='utf-8')
print("\nДанные сохранены в fake_users.csv")

# КАСТОМНЫЙ ПРОВАЙДЕР
from faker.providers import BaseProvider

class CustomProvider(BaseProvider):
    def my_own_data(self):
        return "Собственные данные"

# Регистрация кастомного провайдера
fake.add_provider(CustomProvider)
print(f"Кастомные данные: {fake.my_own_data()}")

# ГЕНЕРАЦИЯ ДАННЫХ ДЛЯ БАЗЫ ДАННЫХ
def generate_customers(n):
    customers = []
    for _ in range(n):
        customer = {
            'customer_id': fake.unique.random_number(digits=8),
            'first_name': fake_ru.first_name(),
            'last_name': fake_ru.last_name(),
            'email': fake.email(),
            'phone': fake_ru.phone_number(),
            'address': fake_ru.street_address(),
            'city': fake_ru.city(),
            'country': fake_ru.country(),
            'postal_code': fake_ru.postcode(),
            'registration_date': fake.date_between(start_date='-5y', end_date='today'),
            'last_login': fake.date_time_this_month(),
            'status': fake.random_element(elements=('active', 'inactive', 'banned')),
            'total_spent': fake.pyfloat(min_value=0, max_value=10000, right_digits=2)
        }
        customers.append(customer)
    return customers

customers = generate_customers(5)
for customer in customers:
    print(f"{customer['first_name']} {customer['last_name']} - {customer['email']}")
