Selenium Documentation

Selenium is a web automation framework for controlling web browsers through programs and automating browser tasks.

Installation:
pip install selenium
# Download WebDriver for your browser
# Chrome: https://chromedriver.chromium.org/
# Firefox: https://github.com/mozilla/geckodriver/releases

Basic Usage:
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.common.keys import Keys
import time

# Initialize driver
driver = webdriver.Chrome()  # or webdriver.Firefox()

# Navigate to page
driver.get('https://www.example.com')

# Find elements
element = driver.find_element(By.ID, 'element_id')
element = driver.find_element(By.NAME, 'element_name')
element = driver.find_element(By.CLASS_NAME, 'class_name')
element = driver.find_element(By.TAG_NAME, 'tag_name')
element = driver.find_element(By.LINK_TEXT, 'link text')
element = driver.find_element(By.PARTIAL_LINK_TEXT, 'partial link text')
element = driver.find_element(By.CSS_SELECTOR, 'css_selector')
element = driver.find_element(By.XPATH, 'xpath')

# Interact with elements
element.click()
element.send_keys('text to input')
element.send_keys(Keys.RETURN)
element.clear()

# Get element properties
text = element.text
value = element.get_attribute('value')
is_displayed = element.is_displayed()
is_enabled = element.is_enabled()

# Wait for elements
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC

# Wait up to 10 seconds for element to be present
wait = WebDriverWait(driver, 10)
element = wait.until(EC.presence_of_element_located((By.ID, 'my_id')))
element = wait.until(EC.element_to_be_clickable((By.ID, 'my_id')))

# Common wait conditions
EC.presence_of_element_located
EC.visibility_of_element_located
EC.element_to_be_clickable
EC.text_to_be_present_in_element
EC.title_contains

Navigation:
driver.get('https://www.example.com')
driver.back()
driver.forward()
driver.refresh()

# JavaScript execution
driver.execute_script('window.scrollTo(0, document.body.scrollHeight);')
driver.execute_script('return document.title;')

Windows and Tabs:
# Open new tab
driver.execute_script('window.open();')
driver.switch_to.window(driver.window_handles[1])

# Close current tab
driver.close()
driver.switch_to.window(driver.window_handles[0])

# Switch to frame
driver.switch_to.frame('frame_name')
driver.switch_to.frame(0)  # by index
driver.switch_to.default_content()  # exit frame

Cookies:
# Get all cookies
cookies = driver.get_cookies()

# Add cookie
driver.add_cookie({'name': 'test', 'value': '123'})

# Delete cookie
driver.delete_cookie('test')
driver.delete_all_cookies()

Screenshots:
# Full page screenshot
driver.save_screenshot('screenshot.png')

# Element screenshot
element = driver.find_element(By.ID, 'my_element')
element.screenshot('element.png')

Headless Mode:
from selenium.webdriver.chrome.options import Options

chrome_options = Options()
chrome_options.add_argument('--headless')
chrome_options.add_argument('--no-sandbox')
chrome_options.add_argument('--disable-dev-shm-usage')

driver = webdriver.Chrome(options=chrome_options)

User Agent:
chrome_options = Options()
chrome_options.add_argument('user-agent=Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36')
driver = webdriver.Chrome(options=chrome_options)

Proxy:
chrome_options = Options()
chrome_options.add_argument('--proxy-server=http://proxy-server:port')
driver = webdriver.Chrome(options=chrome_options)

File Upload:
# Find file input element
file_input = driver.find_element(By.ID, 'file_input')
file_input.send_keys('/path/to/file.txt')

File Download:
chrome_options = Options()
download_dir = '/path/to/download/directory'
prefs = {'download.default_directory': download_dir}
chrome_options.add_experimental_option('prefs', prefs)
driver = webdriver.Chrome(options=chrome_options)

Alerts and Popups:
from selenium.webdriver.common.alert import Alert

# Accept alert
alert = driver.switch_to.alert
alert.accept()

# Dismiss alert
alert.dismiss()

# Get alert text
alert.text

# Send text to alert
alert.send_keys('text')

Actions (Mouse and Keyboard):
from selenium.webdriver.common.action_chains import ActionChains

actions = ActionChains(driver)
actions.move_to_element(element).click().perform()
actions.double_click(element).perform()
actions.context_click(element).perform()  # Right click
actions.drag_and_drop(source, target).perform()

# Keyboard actions
actions.send_keys(Keys.CONTROL, 'a').perform()  # Select all
actions.send_keys(Keys.CONTROL, 'c').perform()  # Copy

Select Dropdown:
from selenium.webdriver.support.ui import Select

select = Select(driver.find_element(By.ID, 'dropdown'))
select.select_by_visible_text('Option 1')
select.select_by_value('option1')
select.select_by_index(0)

# Get selected option
select.first_selected_option.text

# Get all options
for option in select.options:
    print(option.text)

Table Data:
# Find table
table = driver.find_element(By.ID, 'my_table')

# Find all rows
rows = table.find_elements(By.TAG_NAME, 'tr')

for row in rows:
    cells = row.find_elements(By.TAG_NAME, 'td')
    row_data = [cell.text for cell in cells]
    print(row_data)

Error Handling:
try:
    element = driver.find_element(By.ID, 'non_existent')
except NoSuchElementException:
    print('Element not found')

try:
    wait = WebDriverWait(driver, 5)
    element = wait.until(EC.presence_of_element_located((By.ID, 'my_id')))
except TimeoutException:
    print('Element not found within timeout')

Cleanup:
# Always close the driver
driver.quit()  # Closes all windows and quits driver
driver.close()  # Closes current window

Best Practices:
- Use explicit waits instead of time.sleep()
- Use specific locators (ID > Name > CSS > XPath)
- Handle exceptions properly
- Use headless mode for servers
- Clean up resources with driver.quit()
- Use page object pattern for large projects
- Respect robots.txt and website terms
- Don't overload servers with requests
